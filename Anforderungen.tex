\section{Anforderungen an ein Software-Tool zur Sensorbewertung}
\label{sec:Anforderungen}

\begin{itemize}
	\item Sensormodell/Sensorattribute
		\begin{itemize}
			\item FoV (Öffungswinkel, Reichweite)
			\item Datenqualität (Auflösung Genauigkeit, Messdauer)
			\item Arbeitsfrequenz
			\item Umwelteinflüsse (Witterung, Tag/Nacht, Temperatur)
			\item Maße (Größe, Gewicht)
			\item Schnittstellen
		\end{itemize}
	\item Objekte
		\begin{itemize}
			\item Fahrzeuge (KFZ, LKW, Zweirad)
			\item Fußgänger
			\item Geschwindigkeit
		\end{itemize}
	\item Umgebung
		\begin{itemize}
			\item Straße (Spuren, Kreuzung,...)
			\item Objekte (Position, Intention)
			\item Witterung
		\end{itemize}
	\item Bewertung	
\end{itemize}
